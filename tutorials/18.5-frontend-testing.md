## 18.5.2 测试 Pinia Store

### 测试 Auth Store

`auth.js` store 管理用户的登录状态、Token 存储和角色信息。

```javascript
// frontend/src/test/stores/auth.spec.js
import { setActivePinia, createPinia } from 'pinia'
import { useAuthStore } from '../../stores/auth'

describe('Auth Store', () => {
  beforeEach(() => {
    setActivePinia(createPinia())
    localStorage.clear()
  })

  it('初始化状态为未登录', () => {
    const store = useAuthStore()
    expect(store.isAuthenticated).toBe(false)
    expect(store.user).toBeNull()
    expect(store.token).toBeNull()
  })

  it('setUser 设置用户信息后更新认证状态', () => {
    const store = useAuthStore()
    const mockUser = { id: 1, username: 'test', role: 'user' }

    store.setUser(mockUser)

    expect(store.user).toEqual(mockUser)
    expect(store.isAuthenticated).toBe(true)
  })

  it('setToken 保存到 store 和 localStorage', () => {
    const store = useAuthStore()
    const token = 'eyJhbGciOiJIUzI1NiJ9.test-token'

    store.setToken(token)

    expect(store.token).toBe(token)
    expect(localStorage.getItem('token')).toBe(token)
  })

  it('logout 清除所有状态', () => {
    const store = useAuthStore()
    store.setUser({ id: 1, username: 'test', role: 'user' })
    store.setToken('test-token')

    store.logout()

    expect(store.isAuthenticated).toBe(false)
    expect(store.user).toBeNull()
    expect(store.token).toBeNull()
    expect(localStorage.getItem('token')).toBeNull()
  })

  it('isAdmin 对 admin 角色返回 true', () => {
    const store = useAuthStore()
    store.setUser({ id: 1, username: 'admin', role: 'admin' })
    expect(store.isAdmin).toBe(true)
  })

  it('isAdmin 对 user 角色返回 false', () => {
    const store = useAuthStore()
    store.setUser({ id: 2, username: 'user', role: 'user' })
    expect(store.isAdmin).toBe(false)
  })
})

## 18.5.3 测试组件

### 测试 EventCard 组件

`EventCard.vue` 展示活动卡片，包含标题、价格、日期等信息。

```vue
<!-- frontend/src/components/EventCard.vue -->
<template>
  <div class="event-card" @click="goToDetail">
    <div class="card-header">
      <span class="category-tag" :style="{ backgroundColor: categoryColor }">
        {{ categoryLabel }}
      </span>
      <h3>{{ event.title }}</h3>
    </div>
    <div class="card-body">
      <p class="venue">{{ event.venue }}</p>
      <p class="date">{{ formatDate(event.start_date) }}</p>
      <p class="price">¥{{ event.price }}</p>
    </div>
    <div class="card-footer">
      <span :class="['stock', stockClass]">
        {{ event.total_stock - event.sold_count > 0 ?
           `还剩 ${event.total_stock - event.sold_count} 张` :
           '已售罄' }}
      </span>
    </div>
  </div>
</template>

<script setup>
import { computed } from 'vue'
import { useRouter } from 'vue-router'

const props = defineProps({
  event: { type: Object, required: true },
})

const router = useRouter()

const categoryLabels = { concert: '演唱会', theater: '话剧', sports: '体育' }
const categoryLabel = computed(() => categoryLabels[props.event.category] || '其他')

const categoryColors = { concert: '#ff6b6b', theater: '#4ecdc4', sports: '#45b7d1' }
const categoryColor = computed(() => categoryColors[props.event.category] || '#999')

const stockClass = computed(() =>
  props.event.total_stock - props.event.sold_count <= 0 ? 'sold-out' : 'available'
)

function formatDate(dateStr) {
  return new Date(dateStr).toLocaleDateString('zh-CN')
}

function goToDetail() {
  router.push(`/events/${props.event.id}`)
}
</script>
```

### 测试 PaymentCountdown 组件

`PaymentCountdown.vue` 是一个带倒计时功能的支付组件，需要测试计时逻辑。

```javascript
// frontend/src/test/components/PaymentCountdown.spec.js
import { mount } from '@vue/test-utils'
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'
import PaymentCountdown from '../../components/PaymentCountdown.vue'

describe('PaymentCountdown', () => {
  beforeEach(() => {
    vi.useFakeTimers()
  })

  afterEach(() => {
    vi.useRealTimers()
  })

  it('显示初始倒计时', () => {
    const wrapper = mount(PaymentCountdown, {
      props: { expiresAt: Date.now() + 300000 }, // 5分钟后
    })
    // 5:00
    expect(wrapper.text()).toContain('5')
  })

  it('每秒倒计时减少', async () => {
    const now = Date.now()
    const wrapper = mount(PaymentCountdown, {
      props: { expiresAt: now + 300000 },
    })

    vi.advanceTimersByTime(1000) // 前进 1 秒
    await wrapper.vm.$nextTick()

    // 4:59（初始 5:00，过了 1 秒）
    expect(wrapper.text()).toContain('4')
  })

  it('倒计时归零时触发 timeup 事件', async () => {
    const now = Date.now()
    const wrapper = mount(PaymentCountdown, {
      props: { expiresAt: now + 2000 }, // 2秒后过期
    })

    vi.advanceTimersByTime(3000) // 前进 3 秒（超过 2 秒）
    await wrapper.vm.$nextTick()

    expect(wrapper.emitted('timeup')).toBeTruthy()
  })

  it('组件卸载时清除计时器', () => {
    const wrapper = mount(PaymentCountdown, {
      props: { expiresAt: Date.now() + 300000 },
    })
    const clearIntervalSpy = vi.spyOn(global, 'clearInterval')

    wrapper.unmount()

    expect(clearIntervalSpy).toHaveBeenCalled()
  })
})

## 18.5.4 测试 Vue Router 导航守卫

路由守卫控制页面访问权限，需要测试不同角色的访问控制。

```javascript
// frontend/src/test/router/guards.spec.js
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { setActivePinia, createPinia } from 'pinia'
import { useAuthStore } from '../../stores/auth'

describe('Router Navigation Guards', () => {
  beforeEach(() => {
    setActivePinia(createPinia())
  })

  it('未登录用户访问 /orders 被重定向到 /login', () => {
    const store = useAuthStore()
    const from = { path: '/' }
    const to = { path: '/orders' }
    const next = vi.fn()

    // 模拟路由守卫逻辑
    if (!store.isAuthenticated) {
      next('/login')
    } else {
      next()
    }

    expect(next).toHaveBeenCalledWith('/login')
  })

  it('已登录用户访问 /orders 通过', () => {
    const store = useAuthStore()
    store.setUser({ id: 1, username: 'test', role: 'user' })

    const next = vi.fn()
    if (!store.isAuthenticated) {
      next('/login')
    } else {
      next()
    }

    expect(next).toHaveBeenCalledWith()
  })

  it('非 admin 用户访问 /admin 被重定向', () => {
    const store = useAuthStore()
    store.setUser({ id: 1, username: 'user', role: 'user' })

    const next = vi.fn()
    const requiredRole = 'admin'
    const userRole = store.user?.role

    if (requiredRole && userRole !== requiredRole) {
      next('/')
    } else {
      next()
    }

    expect(next).toHaveBeenCalledWith('/')
  })

  it('admin 用户访问 /admin 通过', () => {
    const store = useAuthStore()
    store.setUser({ id: 1, username: 'admin', role: 'admin' })

    const next = vi.fn()
    if (store.user?.role !== 'admin') {
      next('/')
    } else {
      next()
    }

    expect(next).toHaveBeenCalledWith()
  })
})

## 常见问题与排错

| 问题 | 原因 | 解决方法 |
| ------ | ------ | --------- |
| `Cannot find module 'vue'` | 未安装 vue 依赖 | `npm install vue` |
| `useRouter() cannot be used outside of setup` | 组件未正确 stub router | 使用 `vi.mock('vue-router')` mock |
| `[Pinia]: getActivePinia was called with no active Pinia` | 未初始化 Pinia | 在 `beforeEach` 中调用 `setActivePinia(createPinia())` |
| `wrapper.find('.class') not found` | 组件渲染条件不满足 | 检查 props 是否满足渲染条件 |
| `useFakeTimers` 导致测试 hang | 未正确处理 `setInterval` | 确保在 `afterEach` 中调用 `useRealTimers()` |
